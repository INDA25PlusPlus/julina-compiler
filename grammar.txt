
<letter> ::= [A-Z] | [a-z]
<special_character> ::= "_" | <digit>
<digit>  ::= [0-9]

<identifier> ::= <letter> <identifiertail> | <letter>
<identifiertail> ::= <letter> <identifiertail> | <special_character> <identifiertail> | <letter> | <special_character>
<integer> ::= <integer> <digit> | <digit>

<term> ::= <integer> | <identifier> 
<comparison-operation> ::= "<" | ">" | "&&" | "||" | "=="
<condition> ::= <expression> <comparison-operation> <expression> | <expression> <comparsison-operation> <condition>
<if-statement> ::= "if" "(" <condition> ")" "{" <compound-statement> "}"

<compound-statement> ::= <statement> | <statement> <compound-statement>
<statement> ::= <if-statement> | <assignment> | <loop> | <write> 

<operation> ::= "+" | "-"
<expression> ::= <term> <operation> <expression> | <term>
<assignment> ::= <identifier> "=" <expression>

<write> ::= "write" "(" <expression> ")" ";"

<loop> ::= <for-loop> | <while-loop>

<while-loop> ::= "while" "(" <condition> ")" "{" <compound-statement> "}"
